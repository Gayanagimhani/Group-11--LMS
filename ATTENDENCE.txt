---CREATE TABLE ATTENDENCE---

CREATE TABLE ATTENDANCE (
    Reg_no VARCHAR(13) NOT NULL,
    Course_id VARCHAR(8) NOT NULL,
    Date DATE NOT NULL,
    Date_status ENUM('Lecture', 'Quiz', 'Mid', 'End') NOT NULL,
    Session_type ENUM('Theory', 'Practical', 'Both') NOT NULL,
    Attendance_status ENUM('Present', 'Absent') DEFAULT 'Absent',
    Medical_approved ENUM('Yes', 'No', 'Pending') DEFAULT 'No',
    
    FOREIGN KEY (Reg_no) REFERENCES STUDENT(Reg_no)
        ON DELETE CASCADE
        ON UPDATE CASCADE,
    FOREIGN KEY (Course_id) REFERENCES COURSE_UNIT(Course_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);
